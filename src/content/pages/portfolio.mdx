export const projectActionKeys = [
  "live_demo",
  "source_code",
  "published_paper",
];

export const handleProjectImageError = (event) => {
  const placeholderImagePath = "/images/portfolio/placeholder.jpg";
  event.currentTarget.onerror = null;
  event.currentTarget.src = placeholderImagePath;
};

<div className="px-3 sm:px-4 md:px-6 lg:px-0">
  <Grid columns={2} className="items-stretch gap-9 md:gap-12 lg:gap-6 xl:gap-10">
    {projects.map((project) => {
      const actions = projectActionKeys
        .map((key) => {
          const href = project[key];

          if (!href) {
            return null;
          }

          return {
            key,
            href,
            label: portfolioPage.actionLabels[key],
          };
        })
        .filter(Boolean);

      return (
        <Card key={project.title} className="overflow-hidden !p-0">
          <div className="aspect-[16/9] overflow-hidden bg-bg-tertiary">
            <img
              src={project.image}
              alt={`${project.title} preview`}
              className="h-full w-full object-cover"
              loading="lazy"
              onError={handleProjectImageError}
            />
          </div>

          <div className="flex flex-1 flex-col p-7 md:p-8">
            <div className="flex flex-col gap-3 pb-4">
              <div className="flex items-start justify-between gap-4">
                <div>
                  <h3 className="text-[1.35rem] font-bold leading-tight text-text-primary md:text-[1.5rem]">
                    {project.title}
                  </h3>
                </div>
                <span className="shrink-0 rounded-full bg-bg-tertiary px-3 py-1 text-[0.72rem] font-semibold uppercase tracking-[0.08em] text-text-muted">
                  {project.category}
                </span>
              </div>

              <p className="text-[1rem] leading-[1.65] text-text-secondary">
                {project.description}
              </p>

              <TagList className="!mt-0 !pt-0">
                {project.tags.map((tag) => (
                  <Tag key={tag}>{tag}</Tag>
                ))}
              </TagList>
            </div>

            <div className="mt-auto flex flex-wrap items-center gap-x-4 gap-y-2 border-t border-border-subtle pt-4 sm:flex-nowrap">
              {actions.length > 0 ? (
                actions.map((action) => (
                  <a
                    key={action.key}
                    href={action.href}
                    target="_blank"
                    rel="noreferrer"
                    className="whitespace-nowrap text-[0.92rem] font-medium text-accent-primary hover:underline"
                  >
                    {action.label}{" "}
                    <span aria-hidden="true">{portfolioPage.ctaArrow}</span>
                  </a>
                ))
              ) : (
                <span className="text-[0.88rem] font-medium text-text-muted">
                  {portfolioPage.emptyStateLabel}
                </span>
              )}
            </div>
          </div>
        </Card>
      );
    })}
  </Grid>
</div>
